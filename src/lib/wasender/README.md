``` ## Retry Logic The retry system handles transient failures automatically: ```typescript import { withRetry, classifyError, shouldRetry } from '@/lib/wasender'; // Automatic retry with exponential backoff const result = await withRetry( async => { return await sendWhatsAppMessage({ to: phone, body: message }); export async function POST(request: Request) { const signature = request.headers.get('x-wasender-signature'); const body = await request.text; // Verify webhook signature if (!verifyWebhookSignature(body, signature, process.env.WASENDER_WEBHOOK_SECRET)) { return new Response('Invalid signature', { status: 401 }); } const payload = parseWebhookPayload(body); switch (payload.event) { case 'message.delivered': await handleMessageDelivered(payload); break; case 'message.failed': await handleMessageFailed(payload); break; case 'message.read': await handleMessageRead(payload); break; } return new Response('OK', { status: 200 }); } ``` ### Webhook Events | Event | Description | Payload | |-------|-------------|---------| | `message.delivered` | Message delivered to recipient | `{ messageId, timestamp, recipient }` | | `message.failed` | Message delivery failed | `{ messageId, error, retryable }` | | `message.read` | Message read by recipient | `{ messageId, timestamp, recipient }` | | `message.sent` | Message sent from server | `{ messageId, timestamp }` | ## Cron Jobs ### Follow-Up Scheduler Setup Create a Vercel Cron Job or scheduled function: ```typescript // app/api/cron/follow-ups/route.ts import { processFollowUps, getPendingFollowUps } from '@/lib/wasender'; export async function GET(request: Request) { // Verify cron secret const authHeader = request.headers.get('authorization'); if (authHeader !== `Bearer ${process.env.CRON_SECRET}`) { return new Response('Unauthorized', { status: 401 }); } // Process all pending follow-ups const results = await processFollowUps({ batchSize: 50, dryRun: false, }); return Response.json({ processed: results.processed, succeeded: results.succeeded, failed: results.failed, timestamp: new Date.toISOString, }); try { await sendWhatsAppMessage({ to: phone, body: message }); } catch (error) { const classification = classifyError(error); switch (classification.type) { case 'rate_limit': // Queue for later retry await queueMessageForLater(phone, message, classification.suggestedDelayMs); break; case 'invalid_phone': // Mark contact as invalid await markContactInvalid(phone); break; case 'auth_error': // Alert admin await alertAdmin('WhatsApp API authentication failed'); break; case 'network_error': // Retry immediately if (classification.retryable) { await retryWithDelay(phone, message, classification.suggestedDelayMs); } break; default: console.error('Unhandled error:', error); } } ``` ## Rate Limiting The API has the following rate limits: - **Messages per minute**: 100 - **Messages per hour**: 1000 - **Messages per day**: 5000 Use the rate limit headers to manage sending: ```typescript const response = await sendWhatsAppMessage({ to, body });
